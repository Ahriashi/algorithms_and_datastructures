\frametitle{Бинарный поиск}
\begin{lstlisting}[language=C++]
//sorted array 2 5 6 15 21 23 25
int BinarySearch(int* arr, int n, int key) {
  int low = 0;
  int high = n - 1;

  while (low <= high) {
      int mid = (low + high) / 2;
      int midVal = arr[mid];

      if (midVal < key)
          low = mid + 1;
      else if (midVal > key)
          high = mid - 1;
      else
          return mid; // key found
  }
  return -(low + 1);  // key not found.
}
\end{lstlisting}
